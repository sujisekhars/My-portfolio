  <script>

    let goldPerGram = 0, silverPerGram = 0;


    async function fetchLiveRates() {

      try {

        const res = await fetch('https://data-asg.goldprice.org/dbXRates/USD,INR');

        const data = await res.json();

        

        const inr = data.items.find(item => item.curr === 'INR');

        if (!inr) throw new Error();


        const ozToGram = 31.1034768;

        

        // Gold per 10 grams (standard Indian display)

        goldPerGram = inr.xauPrice / ozToGram;

        const gold10g = (goldPerGram * 10).toFixed(2);

        

        // Silver per kg

        silverPerGram = inr.xagPrice / ozToGram;

        const silverKg = (silverPerGram * 1000).toFixed(2);


        // Update DOM

        document.getElementById('gold-rate').textContent = `₹${gold10g}`;

        document.getElementById('gold-oz').textContent = `per oz: ₹${inr.xauPrice.toFixed(0)}`;

        

        document.getElementById('silver-rate').textContent = `₹${silverKg}`;

        document.getElementById('silver-oz').textContent = `per oz: ₹${inr.xagPrice.toFixed(0)}`;


        const time = new Date(data.timestamp * 1000 || Date.now());

        document.getElementById('last-updated').textContent = `• Updated ${time.toLocaleTimeString('en-IN')}`;


        calculatePortfolio();

      } catch(e) {

        console.log("Rate fetch failed (rare)");

      }

    }


    function calculatePortfolio() {

      const goldG = parseFloat(document.getElementById('gold-grams').value) || 0;

      const silverG = parseFloat(document.getElementById('silver-grams').value) || 0;


      const goldVal = (goldG * goldPerGram).toFixed(0);

      const silverVal = (silverG * silverPerGram).toFixed(0);

      const total = (parseFloat(goldVal) + parseFloat(silverVal)).toFixed(0);


      document.getElementById('gold-value').textContent = `Value: ₹${goldVal}`;

      document.getElementById('silver-value').textContent = `Value: ₹${silverVal}`;

      document.getElementById('total-value').textContent = `₹${total}`;

    }


    function saveHoldings() {

      localStorage.setItem('goldGrams', document.getElementById('gold-grams').value);

      localStorage.setItem('silverGrams', document.getElementById('silver-grams').value);

      alert("✅ Holdings saved in your browser!");

    }


    function resetHoldings() {

      if (confirm("Clear all holdings?")) {

        localStorage.clear();

        document.getElementById('gold-grams').value = 0;

        document.getElementById('silver-grams').value = 0;

        calculatePortfolio();

      }

    }


    // Load saved values

    window.onload = () => {

      document.getElementById('gold-grams').value = localStorage.getItem('goldGrams') || 0;

      document.getElementById('silver-grams').value = localStorage.getItem('silverGrams') || 0;

      

      fetchLiveRates();

      setInterval(fetchLiveRates, 30000); // every 30 seconds

      

      // Recalculate on input change

      document.getElementById('gold-grams').addEventListener('input', calculatePortfolio);

      document.getElementById('silver-grams').addEventListener('input', calculatePortfolio);

    };

  </script>
